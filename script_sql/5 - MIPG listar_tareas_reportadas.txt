MODULO MIPG Listar las tareas cargadas en plataforma
----------------------------------------------------
SELECT 
a.id as tareas_id,
a.fecha_realizacion as fecha_realizacion,
a.created_at as fecha_de_reporte,
a.descripcion as descripcion,
a.impacto_kpi,

g.dimension as dimension,
f.politica as policia,

c.name as usuario_reporta,
d.nombre as dependencia_responsable

FROM modulo_mipg_tareas AS a

INNER JOIN modulo_mipg_nivel4s AS b
ON a.nivel4_id= b.id

INNER JOIN users AS c
ON a.user_id = c.id

INNER JOIN entidad_oficinas AS d
ON c.oficina_id = d.id

INNER JOIN modulo_mipg_nivel3s AS e
ON b.nivel3_id = e.id

INNER JOIN modulo_mipg_nivel2s AS f
ON e.nivel2_id = f.id

INNER JOIN modulo_mipg_nivel1s AS g
ON f.nivel1_id = g.id

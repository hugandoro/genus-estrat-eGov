Listar las tareas cargadas en plataforma CON VALORES FINANCIEROS y filtrado por vigencias y dependencia
--------------------------------------------------------------------------------
SELECT 
e.numeral as numero_meta,
a.id as codigo_tarea,
a.descripcion as tarea_descripcion,
a.impacto_kpi as tarea_impacto_kpi,
a.fecha_realizacion as fecha_realizacion,
a.created_at as fecha_reporte,
a.evidencia_pdf as evidencia_pd,
fu1.nombre as fuente1,
a.valor_fuente1 as valor_fuente1,
fu2.nombre as fuente2,
a.valor_fuente2 as valor_fuente2,
fu3.nombre as fuente3,
a.valor_fuente3 as valor_fuente3,
f.nombre as dependencia_responsable

FROM tareas AS a

INNER JOIN plan_accions AS b
ON a.accion_id = b.id

INNER JOIN plan_indicativos AS c
ON b.plan_indicativo_id = c.id

INNER JOIN medicion_indicadors AS d
ON c.indicador_id = d.id

INNER JOIN plan_desarrollo_nivel4s AS e
ON d.nivel4_id = e.id

INNER JOIN entidad_oficinas AS f
ON e.oficina_id = f.id

INNER JOIN general_fuentes AS fu1
ON a.fuente1_id = fu1.id

INNER JOIN general_fuentes AS fu2
ON a.fuente2_id = fu2.id

INNER JOIN general_fuentes AS fu3
ON a.fuente3_id = fu3.id

WHERE (c.vigencia_id = 13)  && (f.id = 20)

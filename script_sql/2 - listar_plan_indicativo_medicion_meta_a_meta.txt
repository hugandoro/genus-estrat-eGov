Listar todo el PLAN INDICATIVO filtrado por una vigencia respectiva, su medicion actual y los datos de la meta respectiva
-------------------------------------------------------------------------------------------------------------------------

SELECT 
c.numeral as numero_meta,
c.nombre  as meta,
b.nombre  as indicador,
d.nombre  as medida,
e.nombre  as tipo_medida,
b.linea_base as linea_base_2019,
b.meta as meta_2023,
b.objetivo as objetivo_cuatrienio,

g.nombre as vigencia,
a.valor as objetivo_vigencia,
a.valor_realizado as valor_realizado,
a.porcentaje_realizado as porcentaje_realizado,
a.rezago as rezago_vigencia,
f.nombre as dependencia_responsable

FROM plan_indicativos AS a

INNER JOIN medicion_indicadors AS b
ON a.indicador_id = b.id

INNER JOIN plan_desarrollo_nivel4s AS c
ON b.nivel4_id = c.id

INNER JOIN medicion_medidas AS d
ON b.medida_id = d.id

INNER JOIN medicion_tipos AS e
ON b.tipo_id = e.id

INNER JOIN entidad_oficinas AS f
ON c.oficina_id = f.id

INNER JOIN general_vigencias AS g
ON a.vigencia_id = g.id

WHERE a.vigencia_id = 12
